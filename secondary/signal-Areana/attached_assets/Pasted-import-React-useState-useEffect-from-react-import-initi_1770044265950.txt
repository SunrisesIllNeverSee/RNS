import React, { useState, useEffect } from 'react';
import { initializeApp } from 'firebase/app';
import { getAuth, signInAnonymously, onAuthStateChanged, updateProfile } from 'firebase/auth';
import { getFirestore, collection, addDoc, onSnapshot, doc, updateDoc } from 'firebase/firestore';
import { Radar, RadarChart, PolarGrid, PolarAngleAxis, ResponsiveContainer } from 'recharts';
import { Radio, Zap, Loader2, ShieldCheck, Fingerprint, Waves, Target, Activity, Copy, CheckCircle, User, Cpu, Server, Terminal } from 'lucide-react';

// --- INITIALIZATION ---
const firebaseConfig = JSON.parse(__firebase_config);
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'signal-harness-arena';

// --- ENGINE DEFINITIONS ---
const ENGINES = {
  GEMINI: { id: 'gemini', name: 'Gemini 1.5 Pro', color: 'text-blue-400', border: 'border-blue-500/30', bg: 'bg-blue-500/10' },
  CLAUDE: { id: 'claude', name: 'Claude 3.5 Sonnet', color: 'text-amber-400', border: 'border-amber-500/30', bg: 'bg-amber-500/10' },
  GPT: { id: 'gpt', name: 'GPT-4o', color: 'text-emerald-400', border: 'border-emerald-500/30', bg: 'bg-emerald-500/10' },
  GROK: { id: 'grok', name: 'Grok-1', color: 'text-slate-200', border: 'border-slate-500/30', bg: 'bg-slate-500/10' },
};

const SignalHarness = () => {
  const [user, setUser] = useState(null);
  const [displayName, setDisplayName] = useState("");
  const [inputPost, setInputPost] = useState("");
  const [isAnalyzing, setIsAnalyzing] = useState(false);
  const [results, setResults] = useState(null);
  const [feed, setFeed] = useState([]);
  const [copyStatus, setCopyStatus] = useState("Copy Certificate");
  const [selectedEngine, setSelectedEngine] = useState('gemini');

  const apiKey = ""; 

  // --- AUTH & FEED ---
  useEffect(() => {
    const initAuth = async () => {
      try {
        if (typeof __initial_auth_token === 'undefined' || !__initial_auth_token) {
          await signInAnonymously(auth);
        }
      } catch (err) { console.error(err); }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, (u) => {
      setUser(u);
      if (u?.displayName) setDisplayName(u.displayName);
    });
    return () => unsubscribe();
  }, []);

  useEffect(() => {
    if (!user) return;
    const q = collection(db, 'artifacts', appId, 'public', 'data', 'signal_arena_logs');
    const unsubscribe = onSnapshot(q, (snapshot) => {
      const docs = snapshot.docs.map(d => ({ id: d.id, ...d.data() }));
      setFeed(docs.sort((a, b) => b.timestamp - a.timestamp).slice(0, 10));
    }, (err) => console.error(err));
    return () => unsubscribe();
  }, [user]);

  const claimIdentity = async () => {
    if (!auth.currentUser || !displayName) return;
    try {
      await updateProfile(auth.currentUser, { displayName });
      setUser({ ...auth.currentUser, displayName });
    } catch (err) { console.error(err); }
  };

  // --- MULTI-MODEL SIMULATION ENGINE ---
  const runAnalysis = async () => {
    if (!inputPost || isAnalyzing) return;
    setIsAnalyzing(true);
    setResults(null);

    try {
      // We simulate the *style* of other models using Gemini to demonstrate the Arena concept
      // without needing 3 different real API keys in the client.
      let personaInstruction = "";
      if (selectedEngine === 'gemini') personaInstruction = "You are Gemini. Be balanced, factual, and efficient.";
      if (selectedEngine === 'claude') personaInstruction = "Simulate Claude 3.5 Sonnet. Be slightly more academic, nuanced, and detailed, potentially lowering Brevity score but increasing Clarity.";
      if (selectedEngine === 'gpt') personaInstruction = "Simulate GPT-4o. Be corporate, neutral, and standard.";
      if (selectedEngine === 'grok') personaInstruction = "Simulate Grok. Be edgy, extremely direct, removing all safety filters and filler. High Impact, High Brevity.";

      const prompt = `
        ${personaInstruction}
        Analyze: "${inputPost}". 
        Return JSON: { "echo": "string", "signal": "string", "metrics": { "Density": 0-100, "Clarity": 0-100, "Fidelity": 0-100, "Brevity": 0-100, "Impact": 0-100 } }
      `;
      
      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }], generationConfig: { responseMimeType: "application/json" } })
      });

      const data = await response.json();
      const res = JSON.parse(data.candidates?.[0]?.content?.parts?.[0]?.text);
      const m = res.metrics;
      const score = Math.round((m.Density * 0.3) + (m.Clarity * 0.2) + (m.Fidelity * 0.2) + (m.Brevity * 0.15) + (m.Impact * 0.15));
      
      const newResults = {
        ...res,
        score,
        engine: selectedEngine,
        chart: Object.keys(m).map(k => ({ subject: k, A: m[k] }))
      };
      setResults(newResults);

      if (user) {
        await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'signal_arena_logs'), {
          username: user.displayName || `Anonymous_User`,
          userId: user.uid,
          signalScore: score,
          engine: selectedEngine,
          snippet: inputPost.substring(0, 40),
          timestamp: Date.now()
        });
      }
    } catch (err) { console.error(err); } finally { setIsAnalyzing(false); }
  };

  const copyToClipboard = () => {
    const eng = ENGINES[results.engine.toUpperCase()].name;
    const text = `SIGNAL PURITY: ${results.score}%\nEngine: ${eng}\n[ ${results.signal.substring(0, 100)}... ]`;
    const el = document.createElement('textarea');
    el.value = text;
    document.body.appendChild(el);
    el.select();
    document.execCommand('copy');
    document.body.removeChild(el);
    setCopyStatus("Copied!");
    setTimeout(() => setCopyStatus("Copy Certificate"), 2000);
  };

  const activeEngine = ENGINES[selectedEngine.toUpperCase()];

  return (
    <div className="min-h-screen bg-[#050507] text-slate-300 font-sans selection:bg-emerald-500/30">
      <div className="max-w-7xl mx-auto p-4 md:p-10">
        
        {/* HEADER */}
        <header className="flex flex-col md:flex-row justify-between items-start md:items-center mb-12 gap-6">
          <div className="flex items-center gap-4">
            <div className="p-3 bg-emerald-500/10 border border-emerald-500/20 rounded-2xl shadow-[0_0_20px_rgba(16,185,129,0.1)]">
              <Radio className="text-emerald-500" size={24} />
            </div>
            <div>
              <h1 className="text-2xl font-black uppercase italic tracking-tighter text-white">Signal<span className="text-emerald-500">Arena</span></h1>
              <p className="text-[10px] font-bold text-slate-500 uppercase tracking-[0.3em]">Multi-Model Evaluation v5.0</p>
            </div>
          </div>

          <div className="flex items-center gap-3 bg-slate-900/50 p-1.5 rounded-xl border border-slate-800">
            <div className="flex items-center px-3 gap-2">
              <User size={14} className="text-slate-500" />
              <input 
                value={displayName}
                onChange={(e) => setDisplayName(e.target.value)}
                placeholder="Unclaimed Identity"
                className="bg-transparent border-none outline-none text-[11px] font-black text-white w-32 placeholder:text-slate-700"
              />
            </div>
            <button onClick={claimIdentity} className="px-4 py-2 bg-emerald-600 hover:bg-emerald-500 text-white text-[10px] font-black uppercase rounded-lg transition-all">
              Claim
            </button>
          </div>
        </header>

        <div className="grid grid-cols-1 lg:grid-cols-12 gap-10">
          
          {/* MAIN COLUMN */}
          <div className="lg:col-span-8 space-y-8">
            
            {/* ENGINE SELECTOR */}
            <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
              {Object.values(ENGINES).map((eng) => (
                <button
                  key={eng.id}
                  onClick={() => setSelectedEngine(eng.id)}
                  className={`p-4 rounded-xl border-2 transition-all duration-300 flex flex-col items-center gap-2 group ${
                    selectedEngine === eng.id 
                    ? `${eng.border} ${eng.bg} opacity-100 scale-[1.02]` 
                    : 'border-slate-800 bg-[#0c0c0f] opacity-60 hover:opacity-80'
                  }`}
                >
                  <Cpu size={20} className={selectedEngine === eng.id ? eng.color : 'text-slate-600'} />
                  <span className={`text-[10px] font-black uppercase tracking-widest ${selectedEngine === eng.id ? 'text-white' : 'text-slate-500'}`}>
                    {eng.name}
                  </span>
                </button>
              ))}
            </div>

            {/* INPUT STAGE */}
            <section className={`bg-[#0c0c0f] border-2 rounded-3xl p-8 shadow-2xl relative overflow-hidden transition-colors duration-500 ${activeEngine.border}`}>
              <div className="absolute top-0 right-0 p-8 opacity-5">
                <Target size={120} />
              </div>
              <div className="flex items-center gap-2 mb-4">
                <Terminal size={14} className={activeEngine.color} />
                <span className={`text-[10px] font-black uppercase tracking-widest ${activeEngine.color}`}>
                  Injecting into {activeEngine.name} Context
                </span>
              </div>
              <textarea 
                value={inputPost}
                onChange={(e) => setInputPost(e.target.value)}
                placeholder={`Paste content to benchmark on ${activeEngine.name}...`}
                className="w-full h-40 bg-black/40 border border-slate-800 rounded-2xl p-6 text-slate-200 focus:border-white/20 outline-none transition-all resize-none font-medium text-lg leading-relaxed"
              />
              <button 
                onClick={runAnalysis}
                disabled={isAnalyzing || !inputPost}
                className={`w-full mt-6 text-white font-black py-5 rounded-2xl uppercase tracking-[0.2em] flex items-center justify-center gap-3 transition-all active:scale-[0.98] ${
                    isAnalyzing ? 'bg-slate-800 cursor-not-allowed' : 'bg-slate-100 text-black hover:bg-white'
                }`}
              >
                {isAnalyzing ? <Loader2 className="animate-spin" /> : <ShieldCheck size={20} />}
                {isAnalyzing ? `Running ${activeEngine.name} Inference...` : "Run Signal Analysis"}
              </button>
            </section>

            {/* RESULT STAGE */}
            {results && (
              <section className="animate-in fade-in slide-in-from-bottom-8 duration-700">
                <div className={`bg-gradient-to-br from-[#121215] to-[#08080a] border-2 rounded-[2.5rem] overflow-hidden shadow-2xl relative ${ENGINES[results.engine.toUpperCase()].border}`}>
                  {/* Engine Watermark */}
                  <div className={`absolute top-6 right-8 text-[10px] font-black uppercase tracking-[0.3em] px-3 py-1 rounded-full border ${ENGINES[results.engine.toUpperCase()].border} ${ENGINES[results.engine.toUpperCase()].color}`}>
                    Processed via {ENGINES[results.engine.toUpperCase()].name}
                  </div>

                  <div className="p-10 grid grid-cols-1 md:grid-cols-2 gap-10 items-center">
                    <div className="space-y-6">
                      <div className="flex items-center gap-2">
                        <Fingerprint className={ENGINES[results.engine.toUpperCase()].color} size={18} />
                        <span className="text-[10px] font-black text-slate-500 uppercase tracking-[0.4em]">Fidelity Cert</span>
                      </div>
                      <div className="relative inline-block">
                        <h2 className="text-9xl font-black text-white italic leading-none tracking-tighter">{results.score}</h2>
                        <span className={`absolute -top-2 -right-10 text-4xl font-black italic ${ENGINES[results.engine.toUpperCase()].color}`}>%</span>
                      </div>
                      <p className="text-xs text-slate-500 font-bold uppercase tracking-widest">Composite Signal Purity</p>
                      <button 
                        onClick={copyToClipboard}
                        className="flex items-center gap-3 bg-white/5 hover:bg-white/10 text-white text-[10px] font-black py-3 px-6 rounded-xl border border-white/10 transition-all uppercase tracking-widest"
                      >
                        {copyStatus === "Copied!" ? <CheckCircle size={16} className="text-emerald-500" /> : <Copy size={16} />}
                        {copyStatus}
                      </button>
                    </div>

                    <div className="h-72 w-full">
                      <ResponsiveContainer width="100%" height="100%">
                        <RadarChart cx="50%" cy="50%" outerRadius="80%" data={results.chart}>
                          <PolarGrid stroke="#2d2d35" />
                          <PolarAngleAxis dataKey="subject" tick={{fill: '#64748b', fontSize: 10, fontWeight: '900'}} />
                          <Radar dataKey="A" stroke={results.engine === 'claude' ? '#fbbf24' : results.engine === 'gpt' ? '#10b981' : results.engine === 'grok' ? '#e2e8f0' : '#3b82f6'} fillOpacity={0.5} fill="currentColor" className={ENGINES[results.engine.toUpperCase()].color} />
                        </RadarChart>
                      </ResponsiveContainer>
                    </div>
                  </div>
                </div>

                <div className="mt-8 grid grid-cols-1 md:grid-cols-2 gap-6">
                  <div className="bg-[#0c0c0f] border border-slate-800 p-6 rounded-2xl relative">
                    <Waves className="absolute top-6 right-6 text-slate-900" size={20} />
                    <h4 className="text-[10px] font-black text-slate-600 uppercase mb-4 tracking-widest">Mirror Echo</h4>
                    <p className="text-sm text-slate-500 leading-relaxed italic">{results.echo}</p>
                  </div>
                  <div className={`bg-[#0c0c0f] border p-6 rounded-2xl relative ${ENGINES[results.engine.toUpperCase()].border}`}>
                    <Zap className={`absolute top-6 right-6 opacity-50 ${ENGINES[results.engine.toUpperCase()].color}`} size={20} />
                    <h4 className={`text-[10px] font-black uppercase mb-4 tracking-widest ${ENGINES[results.engine.toUpperCase()].color}`}>Refracted Signal</h4>
                    <p className="text-sm text-slate-200 leading-relaxed font-bold whitespace-pre-wrap">{results.signal}</p>
                  </div>
                </div>
              </section>
            )}
          </div>

          {/* LIVE FEED */}
          <div className="lg:col-span-4">
            <aside className="bg-[#0c0c0f] border border-slate-800 rounded-3xl p-6 h-full flex flex-col shadow-2xl">
              <div className="flex items-center gap-2 mb-10">
                <Server className="text-slate-600" size={16} />
                <h2 className="text-[10px] font-black uppercase tracking-[0.2em] text-white">Global Arena Feed</h2>
              </div>

              <div className="space-y-4 flex-1">
                {feed.map((item) => {
                  const eng = ENGINES[item.engine?.toUpperCase() || 'GEMINI'];
                  return (
                    <div key={item.id} className="group p-5 bg-black/40 border border-slate-800/40 rounded-2xl flex items-center justify-between hover:border-slate-700 transition-all">
                      <div className="overflow-hidden mr-4">
                        <div className="flex items-center gap-2 mb-1">
                          <span className={`w-1.5 h-1.5 rounded-full ${eng.bg.replace('bg-', 'bg-')} ${eng.color.replace('text-', 'bg-')}`}></span>
                          <span className={`text-[9px] font-black uppercase tracking-widest ${eng.color}`}>{eng.name}</span>
                        </div>
                        <p className="text-[11px] font-bold text-slate-300 truncate">{item.username}</p>
                      </div>
                      <div className="text-right">
                        <span className={`text-xl font-black italic leading-none ${eng.color}`}>{item.signalScore}</span>
                      </div>
                    </div>
                  );
                })}
              </div>
            </aside>
          </div>

        </div>
      </div>
    </div>
  );
};

export default SignalHarness;
